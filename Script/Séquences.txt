CREATE SEQUENCE SALLE_sequence
MINVALUE 1
START WITH 1
INCREMENT BY 1;

CREATE OR REPLACE TRIGGER SALLE_on_insert
  BEFORE INSERT ON SALLE
  FOR EACH ROW
BEGIN
  SELECT SALLE_sequence.nextval
  INTO :new.ID_SALLE
  FROM dual;
END;
